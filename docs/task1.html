## Introduction

This document outlines a demo framework for a cognitive psychology experiment investigating human interaction with Large Language Models (LLMs) in the context of household energy reduction planning. Participants interact with an LLM in different roles (Client, Advisor/Planner) based on provided family energy usage scenarios.

The experiment aims to explore how different reference classes (kWh, $, %) for presenting energy reduction goals and varying LLM roles influence participant planning strategies, advice-giving, and reliance on AI-generated plans.

This demo uses the Purdue GenAI API. You will need to obtain an API key and choose an API endpoint to run the interactive components.

First, ensure you have the necessary Python libraries installed.

  let currentFamilyId = 'davis'; // Can be changed to load different scenarios

  let targetTotalKwh = currentFamily.lastYearTotalKwh - currentFamily.targetReductionKwh;
  let participantPlans = { plan1: {}, plan2: {} }; // To store participant inputs

  function initializeTable() {
    const tableBody = document.getElementById('energy-table-body');
    tableBody.innerHTML = ''; // Clear existing rows
    document.getElementById('target-total').textContent = targetTotalKwh.toLocaleString();

    currentFamily.categories.forEach((cat, index) => {
      const row = tableBody.insertRow();
      row.insertCell(0).textContent = cat.name;
      row.insertCell(1).textContent = cat.lastYear.toLocaleString();
      row.insertCell(2).textContent = cat.stateAvg.toLocaleString();

      // Input for Plan 1
      const plan1Cell = row.insertCell(3);
      const input1 = document.createElement('input');
      input1.type = 'number';
      input1.id = `plan1-cat${index}`;
      input1.min = 0;
      input1.value = cat.lastYear; // Default to last year's value
      input1.style.width = '80px';
      input1.addEventListener('input', () => calculateTotals(1));
      plan1Cell.appendChild(input1);
      participantPlans.plan1[cat.name] = cat.lastYear; // Initialize plan data

      // Input for Plan 2
      const plan2Cell = row.insertCell(4);
      const input2 = document.createElement('input');
      input2.type = 'number';
      input2.id = `plan2-cat${index}`;
      input2.min = 0;
      input2.value = cat.lastYear; // Default to last year's value
      input2.style.width = '80px';
      input2.addEventListener('input', () => calculateTotals(2));
      plan2Cell.appendChild(input2);
      participantPlans.plan2[cat.name] = cat.lastYear; // Initialize plan data
    });
    calculateTotals(1);
    calculateTotals(2);
  }

  function calculateTotals(planNumber) {
    let total = 0;
    const planKey = `plan${planNumber}`;
    const totalElement = document.getElementById(`${planKey}-total`);
    const diffElement = document.getElementById(`${planKey}-diff`);

    currentFamily.categories.forEach((cat, index) => {
      const input = document.getElementById(`${planKey}-cat${index}`);
      const value = parseInt(input.value, 10) || 0;
      participantPlans[planKey][cat.name] = value; // Store current value
      total += value;
    });

    totalElement.textContent = total.toLocaleString();
    const difference = total - targetTotalKwh;
    diffElement.textContent = `${difference >= 0 ? '+' : ''}${difference.toLocaleString()} kWh`;
    diffElement.style.color = Math.abs(difference / targetTotalKwh) < 0.05 ? 'green' : 'red'; // Example: Green if within 5%
  }

  function getPlanAsString(planNumber) {
      let planString = `Plan ${planNumber}:\n`;
      let total = 0;
      const planKey = `plan${planNumber}`;
       currentFamily.categories.forEach((cat, index) => {
          const value = participantPlans[planKey][cat.name] || 0;
          planString += `- ${cat.name}: ${value.toLocaleString()} kWh (Last Year: ${cat.lastYear.toLocaleString()} kWh)\n`;
          total += value;
      });
      planString += `Total: ${total.toLocaleString()} kWh`;
      return planString;
  }


  async function callPurdueGenAI(messages, elementToUpdate) {
    if (!API_KEY || API_KEY === "YOUR_PURDUE_GENAI_API_KEY") {
        elementToUpdate.textContent = "Error: API Key not configured in the script.";
        return "Error: API Key not configured.";
    }
    elementToUpdate.textContent = "AI is thinking...";

    const headers = {
        "Authorization": `Bearer ${API_KEY}`, // [cite: 4]
        "Content-Type": "application/json"
    };
    const body = JSON.stringify({
        model: MODEL_NAME, // [cite: 8]
        messages: messages, // Expects array like [{"role": "user", "content": "..."}]
        stream: false // Set to false for a complete response [cite: 7]
    });

    try {
        const response = await fetch(API_ENDPOINT, { // [cite: 7]
            method: 'POST',
            headers: headers,
            body: body
        });

        if (!response.ok) {
          const errorText = await response.text();
          console.error("API Error:", response.status, errorText);
          throw new Error(`API request failed: ${response.status} ${errorText}`);
        }

        const data = await response.json();

        // Extract content based on OpenAI-compatible structure [cite: 7]
        const aiContent = data.choices && data.choices[0] && data.choices[0].message && data.choices[0].message.content
                           ? data.choices[0].message.content
                           : "Could not parse AI response.";

        elementToUpdate.textContent = aiContent;
        return aiContent; // Return the AI's response content

    } catch (error) {
        console.error('Error calling GenAI API:', error);
        elementToUpdate.textContent = `Error: ${error.message}`;
        return `Error: ${error.message}`;
    }
  }

  }

 function runExperimentCondition(condition) {
    // Hide all condition divs first
    document.getElementById('participant-advisor-ai-client').style.display = 'none';
    document.getElementById('ai-client-ai-planner').style.display = 'none';

    if (condition === 'participantAdvisor') {
        document.getElementById('ai-instructions').textContent = 'You will now act as an advisor to an AI client.';
        document.getElementById('participant-advisor-ai-client').style.display = 'block';
    } else if (condition === 'aiPlanner') {
        document.getElementById('ai-instructions').textContent = 'An AI has also created a plan. You can compare it or ask for advice.';
        document.getElementById('ai-client-ai-planner').style.display = 'block';
        // Pre-emptively ask the AI to generate a plan
        interactWithAI('generatePlanForComparison');
    } else {
         document.getElementById('ai-instructions').textContent = 'AI Interaction section.';
    }
 }

  async function interactWithAI(interactionType) {
    const scenarioDescription = `
      Scenario: The ${currentFamily.name} used ${currentFamily.lastYearTotalKwh.toLocaleString()} kWh last year.
      State average was ${currentFamily.stateAvgTotalKwh.toLocaleString()} kWh.
      Their goal is to reduce usage by ${currentFamily.targetReductionKwh.toLocaleString()} kWh (to a total of ${targetTotalKwh.toLocaleString()} kWh).
      Their usage last year by category was:
      ${currentFamily.categories.map(cat => `- ${cat.name}: ${cat.lastYear.toLocaleString()} kWh`).join('\n')}
    `;

    let userPrompt = "";
    let systemPrompt = ""; // Optional: To guide the AI's persona/role
    let elementToUpdateId = "";

    switch (interactionType) {
        case 'adviseClient':
            systemPrompt = "You are a homeowner client who is somewhat skeptical about changing habits but wants to save energy. Respond to the advisor's plan.";
            userPrompt = `My energy advisor suggested this plan for me:\n${document.getElementById('advisor-input').value}\n\nWhat do you think about this plan?`;
            elementToUpdateId = 'ai-client-response';
            break;

        case 'generatePlanForComparison':
            systemPrompt = "You are an expert AI energy planner. Create one detailed action plan (specific kWh numbers per category) to help the client achieve their energy reduction goal. Explain your reasoning briefly.";
            userPrompt = `Based on this scenario, create an energy reduction plan:\n${scenarioDescription}`;
            elementToUpdateId = 'ai-planner-plan'; // Update the placeholder for the AI's plan
             // No immediate user interaction needed here, just generating the plan
            await callPurdueGenAI([{role: "system", content: systemPrompt}, {role: "user", content: userPrompt}], document.getElementById(elementToUpdateId));
            return; // Exit after generating plan

         case 'comparePlan':
            systemPrompt = "You are an expert AI energy planner. Compare the user's plan with the plan you generated previously. Highlight potential pros and cons of the user's plan compared to yours.";
            const aiPlanText = document.getElementById('ai-planner-plan').textContent; // Get the previously generated plan
            const userPlan1Text = getPlanAsString(1);
            userPrompt = `Here's the scenario:\n${scenarioDescription}\n\nHere is my plan:\n${userPlan1Text}\n\nHere is the plan you generated earlier:\n${aiPlanText}\n\nPlease compare my plan to yours.`;
            elementToUpdateId = 'ai-planner-response';
            break;

        case 'getAdvice':
            systemPrompt = "You are an expert AI energy planner. The user is asking for advice on their plan. Provide constructive feedback or suggestions for improvement.";
             const userPlan1Advice = getPlanAsString(1);
            userPrompt = `Here's the scenario:\n${scenarioDescription}\n\nHere is my plan:\n${userPlan1Advice}\n\nPlease give me advice on how I might improve this plan to meet the goal or make it more realistic.`;
            elementToUpdateId = 'ai-planner-response';
            break;

        default:
            console.error("Unknown interaction type:", interactionType);
            return;
    }

    const messages = [];
    if (systemPrompt) {
        messages.push({ role: "system", content: systemPrompt });
    }
    messages.push({ role: "user", content: userPrompt }); // [cite: 8]

    const elementToUpdate = document.getElementById(elementToUpdateId);
    if (elementToUpdate) {
      await callPurdueGenAI(messages, elementToUpdate);
    } else {
      console.error("Could not find element to update:", elementToUpdateId);
    }
  }

    document.getElementById('scenario-title').textContent = `Scenario: ${currentFamily.name} - Target: ${currentFamily.targetText}`;
     document.getElementById('scenario-details').innerHTML = `
        The ${currentFamily.name} wants to ${currentFamily.targetText} next year.

        Please enter how many kWh should be used next year by each appliance and the total kWh each plan would use. Enter only whole numbers. Try to provide close estimations. You may use a calculator to complete the task.
        <br>
        <em>Note: The family used ${currentFamily.lastYearTotalKwh - currentFamily.stateAvgTotalKwh} more kWh than the average household in their state last year. Target Reduction: ${currentFamily.targetReductionKwh.toLocaleString()} kWh. Target Total kWh: ${targetTotalKwh.toLocaleString()} kWh.</em>
     `;


    initializeTable();
    // Hide AI interaction initially
    document.getElementById('ai-interaction-area').style.display = 'none';

  };
